 182                  return $this->row_data[$n];
 183              }
 184              // reset the $n variable if the result was not achieved
 185              $n = 0;
 186          }

 188          if ($type == 'object') return $this->row_object($n);
 189          else if ($type == 'array') return $this->row_array($n);
