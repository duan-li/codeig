 162  	private function _set_platform()
 163      {
 164          if (is_array($this->platforms) AND count($this->platforms) > 0)
 165          {
 166              foreach ($this->platforms as $key => $val)
 167              {
 168                  if (preg_match("|".preg_quote($key)."|i", $this->agent))
 169                  {
 170                      $this->platform = $val;
 171                      return TRUE;
 172                  }
 173              }
 174          }
