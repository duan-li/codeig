 106      /**
 107       * Encode
 108       *
 109       * Encodes the message string using bitwise XOR encoding.
 110       * The key is combined with a random hash, and then it
 111       * too gets converted using XOR. The whole thing is then run
 112       * through mcrypt using the randomized key. The end result
 113       * is a double-encrypted message string that is randomized
 114       * with each call to this function, even if the supplied
 115       * message and key are the same.
 116       *
 117       * @access    public
 118       * @param    string    the string to encode
 119       * @param    string    the key
 120       * @return    string
 121       */
 122  	function encode($string, $key = '')
 123      {
 124          $key = $this->get_key($key);
 125          $enc = $this->mcrypt_encode($string, $key);
 126  
 127          return base64_encode($enc);
 128      }

 142  	function decode($string, $key = '')
 143      {
 144          $key = $this->get_key($key);
 145  
 146          if (preg_match('/[^a-zA-Z0-9\/\+=]/', $string))
 147          {
 148              return FALSE;
 149          }
 150  
 151          $dec = base64_decode($string);
 152  
 153          if (($dec = $this->mcrypt_decode($dec, $key)) === FALSE)
 154          {
 155              return FALSE;
 156          }
 157  
 158          return $dec;

 292          if ($init_size > strlen($data))

 294              return FALSE;
 295          }
 296  
 297          $init_vect = substr($data, 0, $init_size);
 298          $data = substr($data, $init_size);
 299          return rtrim(mcrypt_decrypt($this->_get_cipher(), $key, $data, $this->_get_mode(), $init_vect), "\0");
 300      }

 347  	function _remove_cipher_noise($data, $key)

 349          $keyhash = $this->hash($key);
 350          $keylen = strlen($keyhash);
 351          $str = '';
 352  
 353          for ($i = 0, $j = 0, $len = strlen($data); $i < $len; ++$i, ++$j)
 354          {
 355              if ($j >= $keylen)
 356              {
 357                  $j = 0;
 358              }
 359  
 360              $temp = ord($data[$i]) - ord($keyhash[$j]);
 361  
 362              if ($temp < 0)
 363              {
 364                  $temp = $temp + 256;
 365              }
 366  
 367              $str .= chr($temp);
 368          }
 369  
 370          return $str;

 469       *

 471       * @param    string
 472       * @return    string
 473       */
 474  	function sha1($str)
 475      {
 476          if ( ! function_exists('sha1'))
 477          {
 478              if ( ! function_exists('mhash'))
 479              {
 480                  require_once(BASEPATH.'libraries/Sha1.php');
 481                  $SH = new CI_SHA;
 482                  return $SH->generate($str);
 483              }
 484              else
 485              {
 486                  return bin2hex(mhash(MHASH_SHA1, $str));
 487              }
 488          }
 489          else
 490          {
 491              return sha1($str);
