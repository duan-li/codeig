 129      /**
 130       * Fetch the current session data if it exists
 131       *
 132       * @access    public
 133       * @return    bool
 134       */
 135  	function sess_read()
 136      {
 137          // Fetch the cookie
 138          $session = $this->CI->input->cookie($this->sess_cookie_name);
 139  
 140          // No cookie?  Goodbye cruel world!...
 141          if ($session === FALSE)
 142          {
 143              log_message('debug', 'A session cookie was not found.');
 144              return FALSE;
 145          }
 146  
 147          // HMAC authentication
 148          $len = strlen($session) - 40;
 149  
 150          if ($len <= 0)
 151          {
 152              log_message('error', 'Session: The session cookie was not signed.');
 153              return FALSE;
 154          }
 155  
 156          // Check cookie authentication
 157          $hmac = substr($session, $len);
 158          $session = substr($session, 0, $len);
 159  
 160          // Time-attack-safe comparison
 161          $hmac_check = hash_hmac('sha1', $session, $this->encryption_key);
 162          $diff = 0;
 163  
 164          for ($i = 0; $i < 40; $i++)
 165          {
 166              $xor = ord($hmac[$i]) ^ ord($hmac_check[$i]);
 167              $diff |= $xor;
 168          }
 169  
 170          if ($diff !== 0)
 171          {
 172              log_message('error', 'Session: HMAC mismatch. The session cookie data did not match what was expected.');
 173              $this->sess_destroy();
 174              return FALSE;
 175          }
 176  
 177          // Decrypt the cookie data
 178          if ($this->sess_encrypt_cookie == TRUE)
 179          {
 180              $session = $this->CI->encrypt->decode($session);
 181          }
 182  
 183          // Unserialize the session array
 184          $session = $this->_unserialize($session);
 185  
 186          // Is the session data we unserialized an array with the correct format?
 187          if ( ! is_array($session) OR ! isset($session['session_id']) OR ! isset($session['ip_address']) OR ! isset($session['user_agent']) OR ! isset($session['last_activity']))
 188          {
 189              $this->sess_destroy();
 190              return FALSE;
 191          }
 192  
 193          // Is the session current?
 194          if (($session['last_activity'] + $this->sess_expiration) < $this->now)
 195          {
 196              $this->sess_destroy();
 197              return FALSE;
 198          }
 199  
 200          // Does the IP Match?
 201          if ($this->sess_match_ip == TRUE AND $session['ip_address'] != $this->CI->input->ip_address())
 202          {
 203              $this->sess_destroy();
 204              return FALSE;
 205          }
 206  
 207          // Does the User Agent Match?
 208          if ($this->sess_match_useragent == TRUE AND trim($session['user_agent']) != trim(substr($this->CI->input->user_agent(), 0, 120)))
 209          {
 210              $this->sess_destroy();
 211              return FALSE;
 212          }
 213  
 214          // Is there a corresponding session in the DB?
 215          if ($this->sess_use_database === TRUE)
 216          {
 217              $this->CI->db->where('session_id', $session['session_id']);
 218  
 219              if ($this->sess_match_ip == TRUE)
 220              {
 221                  $this->CI->db->where('ip_address', $session['ip_address']);
 222              }
 223  
 224              if ($this->sess_match_useragent == TRUE)
 225              {
 226                  $this->CI->db->where('user_agent', $session['user_agent']);
 227              }
 228  
 229              $query = $this->CI->db->get($this->sess_table_name);
 230  
 231              // No result?  Kill it!
 232              if ($query->num_rows() == 0)
 233              {
 234                  $this->sess_destroy();
 235                  return FALSE;
 236              }
 237  
 238              // Is there custom data?  If so, add it to the main session array
 239              $row = $query->row();
 240              if (isset($row->user_data) AND $row->user_data != '')
 241              {
 242                  $custom_data = $this->_unserialize($row->user_data);
 243  
 244                  if (is_array($custom_data))
 245                  {
 246                      foreach ($custom_data as $key => $val)
 247                      {
 248                          $session[$key] = $val;
 249                      }
 250                  }
 251              }
 252          }
 253  
 254          // Session is valid!
 255          $this->userdata = $session;
 256          unset($session);
 257  
 258          return TRUE;
 259      }

 263      /**
 264       * Write the session data
 265       *
 266       * @access    public
 267       * @return    void
 268       */
 269  	function sess_write()
 270      {
 271          // Are we saving custom data to the DB?  If not, all we do is update the cookie
 272          if ($this->sess_use_database === FALSE)
 273          {
 274              $this->_set_cookie();
 275              return;
 276          }
 277  
 278          // set the custom userdata, the session data we will set in a second
 279          $custom_userdata = $this->userdata;
 280          $cookie_userdata = array();
 281  
 282          // Before continuing, we need to determine if there is any custom data to deal with.
 283          // Let's determine this by removing the default indexes to see if there's anything left in the array
 284          // and set the session data while we're at it
 285          foreach (array('session_id','ip_address','user_agent','last_activity') as $val)
 286          {
 287              unset($custom_userdata[$val]);
 288              $cookie_userdata[$val] = $this->userdata[$val];
 289          }
 290  
 291          // Did we find any custom data?  If not, we turn the empty array into a string
 292          // since there's no reason to serialize and store an empty array in the DB
 293          if (count($custom_userdata) === 0)
 294          {
 295              $custom_userdata = '';
 296          }
 297          else
 298          {
 299              // Serialize the custom data array so we can store it
 300              $custom_userdata = $this->_serialize($custom_userdata);
 301          }
 302  
 303          // Run the update query
 304          $this->CI->db->where('session_id', $this->userdata['session_id']);
 305          $this->CI->db->update($this->sess_table_name, array('last_activity' => $this->userdata['last_activity'], 'user_data' => $custom_userdata));
 306  
 307          // Write the cookie.  Notice that we manually pass the cookie data array to the
 308          // _set_cookie() function. Normally that function will store $this->userdata, but
 309          // in this case that array contains custom data, which we do not want in the cookie.
 310          $this->_set_cookie($cookie_userdata);
 311      }

 315      /**
 316       * Create a new session
 317       *
 318       * @access    public
 319       * @return    void
 320       */
 321  	function sess_create()
 322      {
 323          $sessid = '';
 324          while (strlen($sessid) < 32)
 325          {
 326              $sessid .= mt_rand(0, mt_getrandmax());
 327          }
 328  
 329          // To make the session ID even more secure we'll combine it with the user's IP
 330          $sessid .= $this->CI->input->ip_address();
 331  
 332          $this->userdata = array(
 333                              'session_id'    => md5(uniqid($sessid, TRUE)),
 334                              'ip_address'    => $this->CI->input->ip_address(),
 335                              'user_agent'    => substr($this->CI->input->user_agent(), 0, 120),
 336                              'last_activity'    => $this->now,
 337                              'user_data'        => ''
 338                              );
 339  
 340  
 341          // Save the data to the DB if needed
 342          if ($this->sess_use_database === TRUE)
 343          {
 344              $this->CI->db->query($this->CI->db->insert_string($this->sess_table_name, $this->userdata));
 345          }
 346  
 347          // Write the cookie
 348          $this->_set_cookie();
 349      }

 354       * Update an existing session
 355       *
 356       * @access    public
 357       * @return    void
 358       */
 359  	function sess_update()
 360      {
 361          // We only update the session every five minutes by default
 362          if (($this->userdata['last_activity'] + $this->sess_time_to_update) >= $this->now)
 363          {
 364              return;
 365          }
 366  
 367          // Save the old session id so we know which record to
 368          // update in the database if we need it
 369          $old_sessid = $this->userdata['session_id'];
 370          $new_sessid = '';
 371          while (strlen($new_sessid) < 32)
 372          {
 373              $new_sessid .= mt_rand(0, mt_getrandmax());
 374          }
 375  
 376          // To make the session ID even more secure we'll combine it with the user's IP
 377          $new_sessid .= $this->CI->input->ip_address();
 378  
 379          // Turn it into a hash
 380          $new_sessid = md5(uniqid($new_sessid, TRUE));
 381  
 382          // Update the session data in the session data array
 383          $this->userdata['session_id'] = $new_sessid;
 384          $this->userdata['last_activity'] = $this->now;
 385  
 386          // _set_cookie() will handle this for us if we aren't using database sessions
 387          // by pushing all userdata to the cookie.
 388          $cookie_data = NULL;
 389  
 390          // Update the session ID and last_activity field in the DB if needed
 391          if ($this->sess_use_database === TRUE)
 392          {
 393              // set cookie explicitly to only have our session data
 394              $cookie_data = array();
 395              foreach (array('session_id','ip_address','user_agent','last_activity') as $val)
 396              {
 397                  $cookie_data[$val] = $this->userdata[$val];
 398              }
 399  
 400              $this->CI->db->query($this->CI->db->update_string($this->sess_table_name, array('last_activity' => $this->now, 'session_id' => $new_sessid), array('session_id' => $old_sessid)));
 401          }
 402  
 403          // Write the cookie
 404          $this->_set_cookie($cookie_data);
 405      }

 430                      $this->cookie_domain,
 431                      0
 432                  );
 433  
 434          // Kill session data
 435          $this->userdata = array();

 694       * Serialize an array
 695       *
 696       * This function first converts any slashes found in the array to a temporary
 697       * marker, so when it gets unserialized the slashes will be preserved
 698       *
 699       * @access    private

 701       * @return    string

 703  	function _serialize($data)
 704      {

 708              {
 709                  if (is_string($val))

 712                  }
 713              }

 718              {

 720              }
 721          }
